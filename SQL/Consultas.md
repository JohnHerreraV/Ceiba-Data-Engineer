Desembolso diario

SELECT 
    ORIGINATION_DATE, 
    COUNT(LOAN_ID) AS TOTAL_CREDITOS_DESEMBOLSADOS,
    SUM(PRINCIPAL_AMOUNT) AS MONTO_TOTAL_DESEMBOLSADO,
    -- Calculamos el promedio por crédito ese día para ver el ticket promedio
    ROUND(AVG(PRINCIPAL_AMOUNT), 2) AS TICKET_PROMEDIO_DIA
FROM CEIBA.RAW_FINTRUST.LOANS
GROUP BY 1
ORDER BY 1 DESC;

Recaudo diario

SELECT 
    PAYMENT_DATE,
    COUNT(PAYMENT_ID) AS CANTIDAD_TRANSACCIONES,
    SUM(PAYMENT_AMOUNT) AS MONTO_RECAUDADO_TOTAL,
    -- Desglosamos por canal para entender por dónde entra más dinero
    PAYMENT_CHANNEL,
    -- Calculamos el recaudo promedio por transacción
    ROUND(AVG(PAYMENT_AMOUNT), 2) AS TICKET_PROMEDIO_RECAUDO
FROM CEIBA.RAW_FINTRUST.PAYMENTS
WHERE PAYMENT_STATUS = 'CONFIRMED'
GROUP BY 1, 4
ORDER BY 1 DESC, 3 DESC;

Cartera por Cohorte

SELECT 
    -- 1. Definimos la cohorte (Mes de originación)
    DATE_TRUNC('MONTH', ORIGINATION_DATE) AS COHORTE_MES,
    
    -- 2. Métricas de Volumen
    COUNT(LOAN_ID) AS CANTIDAD_CREDITOS,
    SUM(LOAN_AMOUNT) AS MONTO_ORIGINADO,
    
    -- 3. Métricas de Estado de Cartera (Saldos)
    SUM(OUTSTANDING_BALANCE) AS SALDO_PENDIENTE_TOTAL,
    
    -- 4. Segmentación de Riesgo
    SUM(CASE WHEN FINANCIAL_STATUS_CALC = 'EN MORA' THEN OUTSTANDING_BALANCE ELSE 0 END) AS SALDO_EN_MORA,
    SUM(CASE WHEN FINANCIAL_STATUS_CALC = 'PAGADO' THEN 1 ELSE 0 END) AS CREDITOS_FINALIZADOS,
    
    -- 5. Indicadores Clave (KPIs) de la Cohorte
    ROUND(
        (SUM(CASE WHEN FINANCIAL_STATUS_CALC = 'EN MORA' THEN OUTSTANDING_BALANCE ELSE 0 END) / 
        NULLIF(SUM(OUTSTANDING_BALANCE), 0)) * 100, 2
    ) AS INDICE_MORA_COHORTE_PCT,
    
    ROUND(AVG(LOAN_AMOUNT), 2) AS TICKET_PROMEDIO_COHORTE

FROM CEIBA.RAW_FINTRUST.ANALYSIS_LOAN_PERFORMANCE
GROUP BY 1
ORDER BY 1 DESC;

Top 10 por atrasos

SELECT 
    CUSTOMER_NAME,
    CUSTOMER_SEGMENT,
    LOAN_ID,
    PRODUCT_TYPE,
    -- Prioridad 1: Cantidad de cuotas que debe
    OVERDUE_INSTALLMENTS_COUNT AS CUOTAS_VENCIDAS,
    -- Prioridad 2: Cuánto dinero representa ese atraso
    OUTSTANDING_BALANCE AS SALDO_PENDIENTE,
    -- Calculamos el porcentaje del crédito que aún no se ha pagado
    ROUND((OUTSTANDING_BALANCE / NULLIF(LOAN_AMOUNT, 0)) * 100, 2) AS EXPOSICION_PCT,
    -- Clasificación para el equipo de cobranza
    CASE 
        WHEN OVERDUE_INSTALLMENTS_COUNT >= 3 THEN 'CRÍTICO (CERRAR CUENTA)'
        WHEN OVERDUE_INSTALLMENTS_COUNT BETWEEN 1 AND 2 THEN 'PREVENTIVO (LLAMADA)'
        ELSE 'SIN MORA'
    END AS ACCION_RECOMENDADA
FROM CEIBA.RAW_FINTRUST.ANALYSIS_LOAN_PERFORMANCE
WHERE FINANCIAL_STATUS_CALC = 'EN MORA'
ORDER BY CUOTAS_VENCIDAS DESC, SALDO_PENDIENTE DESC
LIMIT 10;
